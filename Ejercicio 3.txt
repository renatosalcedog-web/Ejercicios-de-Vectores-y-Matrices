#include <iostream>
#include <vector>

using namespace std;

int main() {
    int n, productoEscalar = 0;

    cout << "Ingrese el tamaño de los vectores: ";
    cin >> n;

    vector<int> A(n);
    vector<int> B(n);

    cout << "\nIngrese los elementos del vector A:" << endl;
    for (int i = 0; i < n; i++) {
        cout << "A[" << i << "] = ";
        cin >> A[i];
    }

    cout << "\nIngrese los elementos del vector B:" << endl;
    for (int i = 0; i < n; i++) {
        cout << "B[" << i << "] = ";
        cin >> B[i];
    }

    for (int i = 0; i < n; i++) {
        productoEscalar = productoEscalar + (A[i] * B[i]);
    }

    cout << "\nEl producto escalar entre los vectores es: "
         << productoEscalar << endl;

    return 0;
}
