// EJERCICIO 5
// Escribir un programa que calcule el vector resultante de sumar a cada una de las componentes de un vector dado, un número real.

#include <iostream>

using namespace std;

int main() {
    int tam;

    cout << "Ingrese la cantidad de elementos del vector: ";
    cin >> tam;

    float vectorDatos[tam];
    float valor;

    cout << "\n--- INGRESO DEL VECTOR ---\n";
    for (int i = 0; i < tam; i++) {
        cout << "Elemento [" << i << "]: ";
        cin >> vectorDatos[i];
    }

    cout << "\nIngrese el numero que desea sumar a cada elemento: ";
    cin >> valor;

    cout << "\nVector resultante:\n";
    for (int i = 0; i < tam; i++) {
        vectorDatos[i] += valor;
        cout << "Elemento [" << i << "] = " << vectorDatos[i] << endl;
    }

    return 0;
}
